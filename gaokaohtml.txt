function loadJquery(callback) {
	var addjqsrc = true;
	if(document.getElementsByTagName("script").length > 0){
		var scriptstr  = document.getElementsByTagName("script");
		for(var itest = 0;itest<scriptstr.length;itest++){
			if(scriptstr[itest].getAttribute("src") && scriptstr[itest].getAttribute("src").indexOf("jquery.min.js") != -1){
				addjqsrc=false;
				callback();
				break;
			}
		}
	}
	if(addjqsrc){	
		var jqtest = document.createElement('script');
		jqtest.type = 'text/javascript';
		jqtest.async = true;
		jqtest.src = ('https:' == document.location.protocol ? 'https://' : 'http://') + 't1.chei.com.cn/common/jquery/1.8.3/jquery.min.js';
		if(jqtest.readyState){
			//IE
			jqtest.onreadystatechange=function(){
				if(jqtest.readyState=="loaded"||jqtest.readyState=="complete"){
					jqtest.onreadystatechange=null;
					callback();
				}
			};
		}else{
			//其他浏览器
			jqtest.onload=function(){
				callback();
			};
		}
		if(document.getElementsByTagName("script").length > 0){
			document.getElementsByTagName('script')[0].parentNode.insertBefore(jqtest, document.getElementsByTagName('script')[0]);
		}else{
			document.getElementsByTagName("head")[0].appendChild(jqtest);
		}
	}
}

			}else if(d=="2"){
				showMessage("已经收藏该专业",obj);
				$(obj).removeClass("new").addClass("remove");
			}else if(d=="3"){
				showMessage("超过可收藏院校总数",obj);
			}else if(d=="4"){
				showMessage("超过该院校可收藏专业总数",obj);
			}else if(d=="4"){
				showMessage("超过可收藏总数",obj);
			}else if(d=="10"){
				showMessage("只能收藏生源地和您所在地相同的专业",obj);
			}
		});
	})
})
function resizeH(){
	var lh = $("#leftHeight").height() - 35;
	var rh = $("#rightHeight").height();
	var addh = lh - rh;
	if(addh > 0){
		$("#rightHeight").height($("#rightHeight").height() + addh);
	}else{
		$("#leftHeight").height($("#leftHeight").height() - addh);
	}
}
function showMessage(msg,ele){
    var msgDiv = $('.respondMessage');
    msgDiv.stop(true, true);
    msgDiv.html(msg);
	msgDiv.css({ top: 0, left: 0 });
	var top = ele.offset().top - 42;
	var left = ele.offset().left - 60;
	msgDiv.offset({ top: top, left: left });;
    msgDiv.fadeIn(300).delay(1000).fadeOut(300);
}
